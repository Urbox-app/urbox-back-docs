const { Router } = require('express');

const registerController = require('../controllers/registerController');
const validateFirebaseToken = require('../middlewares/validateFirebaseToken');

/**
 * Enrutador para las operaciones relacionadas con los registros.
 *
 * @namespace RegisterRouter
 */
const router = Router();

const PATH = '/api/register';

/**
 * Ruta para la creación de un nuevo registro.
 * Se requiere un token de Firebase para acceder.
 *
 * @name POST /api/register
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para la creación de registros.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para la creación de registros.
 */
router.post(PATH, validateFirebaseToken, registerController.createRegister);

/**
 * Ruta para obtener la lista de registros.
 * Se requiere un token de Firebase para acceder.
 *
 * @name GET /api/register
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para obtener la lista de registros.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para obtener la lista de registros.
 */
router.get(PATH, validateFirebaseToken, registerController.getRegisters);

/**
 * Ruta para eliminar un registro.
 * Se requiere un token de Firebase para acceder.
 *
 * @name DELETE /api/register
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para eliminar un registro.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para eliminar un registro.
 */
router.delete(PATH, validateFirebaseToken, registerController.deleteRegister);

/**
 * Ruta para actualizar un registro.
 * Se requiere un token de Firebase para acceder.
 *
 * @name PATCH /api/register
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para actualizar un registro.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para actualizar un registro.
 */
router.patch(PATH, validateFirebaseToken, registerController.updateRegister);

/**
 * Ruta para filtrar registros por fecha, tipo, etc.
 * Se requiere un token de Firebase para acceder.
 *
 * @name POST /api/register/filters
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para filtrar registros.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para filtrar registros.
 */
router.post(
  `${PATH}/filters`,
  validateFirebaseToken,
  registerController.filterRegisters
);

/**
 * Ruta para obtener datos del panel de control.
 * Se requiere un token de Firebase para acceder.
 *
 * @name GET /api/register/dashboard
 * @function
 * @memberof RegisterRouter
 * @inner
 * @param {string} path - Ruta relativa para obtener datos del panel de control.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para obtener datos del panel de control.
 */
router.get(
  `${PATH}/dashboard`,
  validateFirebaseToken,
  registerController.getDashboardData
);

module.exports = router;
