const express = require('express');

const router = express.Router();
const userController = require('../controllers/tenantController');
const validateFirebaseToken = require('../middlewares/validateFirebaseToken');

/**
 * Ruta principal para operaciones relacionadas con el usuario.
 * Se requiere un token de Firebase para acceder.
 *
 * @name POST /api/user
 * @function
 * @memberof module:routes/user
 * @inner
 * @param {string} path - Ruta relativa para la creación de usuarios.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para la creación de usuarios.
 */
router.post('/api/user', validateFirebaseToken, userController.createUser);

/**
 * Ruta para agregar un nuevo inquilino al usuario existente.
 * Se requiere un token de Firebase para acceder.
 *
 * @name POST /api/user/add
 * @function
 * @memberof module:routes/user
 * @inner
 * @param {string} path - Ruta relativa para agregar un inquilino al usuario.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para agregar un inquilino al usuario.
 */
router.post('/api/user/add', validateFirebaseToken, userController.addTenant);

/**
 * Ruta para obtener información del cliente.
 * Se requiere un token de Firebase para acceder.
 *
 * @name GET /api/user
 * @function
 * @memberof module:routes/user
 * @inner
 * @param {string} path - Ruta relativa para obtener información del cliente.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para obtener información del cliente.
 */
router.get('/api/initialize', validateFirebaseToken, userController.initialize);

/**
 * Ruta para actualizar la información del cliente.
 * Se requiere un token de Firebase para acceder.
 *
 * @name PATCH /api/user
 * @function
 * @memberof module:routes/user
 * @inner
 * @param {string} path - Ruta relativa para actualizar información del cliente.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para actualizar información del cliente.
 */
router.patch(
  '/api/user/changeTenant',
  validateFirebaseToken,
  userController.updateCurrentTenant
);

/**
 * Ruta para actualizar la información de un inquilino.
 * Se requiere un token de Firebase para acceder.
 *
 * @name PATCH /api/user/update
 * @function
 * @memberof module:routes/user
 * @inner
 * @param {string} path - Ruta relativa para actualizar información de un inquilino.
 * @param {Function} middleware - Middleware para validar el token de Firebase.
 * @param {Function} controller - Controlador para actualizar información de un inquilino.
 */
router.patch(
  '/api/user/update',
  validateFirebaseToken,
  userController.updateTenant
);

module.exports = router;
