/**
 * Calcula los totales de los pedidos y los gastos.
 * @function
 * @param {Object[]} arr - Lista de pedidos.
 * @param {Object[]} expenses - Lista de gastos.
 * @returns {Object} Objeto con los totales calculados.
 * @throws {string} Mensaje si el parÃ¡metro arr no es un arreglo.
 * @example
 * const orders = [
 *   { bill: { total: 100, must: 50 } },
 *   { bill: { total: 150, must: 75 } }
 * ];
 * const expenses = [
 *   { value: 20 },
 *   { value: 30 }
 * ];
 * const totals = getTotals(orders, expenses);
 * // Resultado:
 * // {
 * //   total: 250,
 * //   cash: 150,
 * //   must: 125
 * // }
 */
const getTotals = (arr, expenses) => {
  if (arr && !arr.push) return 'element is not an array';

  let total = 0;
  const initialValue = 0;

  const bills = arr.map((doc) => ({ bill: doc.bill }));
  const totalExpenses = expenses.reduce(
    (accumulator, currentValue) => accumulator + currentValue.value,
    initialValue
  );

  const must = bills.reduce(
    (accumulator, currentValue) => accumulator + currentValue.bill.must,
    initialValue
  );

  bills.forEach((bill) => {
    total += parseInt(bill.bill.total, 10);
  });

  const cash = total - totalExpenses - must;

  return { total, cash, must };
};

module.exports = getTotals;
