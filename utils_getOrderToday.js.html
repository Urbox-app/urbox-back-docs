const moment = require('moment');

/**
 * Filtra los pedidos que fueron entregados hoy.
 * @function
 * @param {Object[]} arr - Lista de pedidos.
 * @returns {Object[]} Lista de pedidos entregados hoy.
 * @throws {string} Mensaje si el parÃ¡metro arr no es un arreglo.
 * @example
 * const orders = [
 *   { bill: { delivery: '2023-08-08' } },
 *   { bill: { delivery: '2023-08-07' } },
 *   { bill: { delivery: '2023-08-08' } }
 * ];
 * const todayOrders = getOrderToday(orders);
 * // Resultado:
 * // [
 * //   { bill: { delivery: '2023-08-08' } },
 * //   { bill: { delivery: '2023-08-08' } }
 * // ]
 */
const getOrderToday = (arr) => {
  if (arr && !arr.push) return 'element is not an array';
  const orderToday = arr.filter(
    (i) => i.bill.delivery === moment().toISOString()
  );
  return orderToday;
};

module.exports = getOrderToday;
