const express = require('express');

const router = express.Router();
const expenseController = require('../controllers/expenseController');
const validateFirebaseToken = require('../middlewares/validateFirebaseToken');

/**
 * Rutas relacionadas con la gesti√≥n de gastos.
 *
 * @namespace ExpenseRouter
 */
const PATCH = '/api/expenses';

/**
 * Ruta para crear un nuevo gasto.
 *
 * @name POST/api/expenses
 * @function
 * @memberof ExpenseRouter
 * @inner
 * @param {Object} req - Objeto de solicitud de Express.
 * @param {Object} res - Objeto de respuesta de Express.
 */
router.post(PATCH, validateFirebaseToken, expenseController.createExpense);

/**
 * Ruta para obtener la lista de gastos.
 *
 * @name GET/api/expenses
 * @function
 * @memberof ExpenseRouter
 * @inner
 * @param {Object} req - Objeto de solicitud de Express.
 * @param {Object} res - Objeto de respuesta de Express.
 */
router.get(PATCH, validateFirebaseToken, expenseController.getExpenses);

/**
 * Ruta para eliminar un gasto.
 *
 * @name DELETE/api/expenses
 * @function
 * @memberof ExpenseRouter
 * @inner
 * @param {Object} req - Objeto de solicitud de Express.
 * @param {Object} res - Objeto de respuesta de Express.
 */
router.delete(PATCH, validateFirebaseToken, expenseController.deleteExpense);

/**
 * Ruta para actualizar un gasto.
 *
 * @name PATCH/api/expenses
 * @function
 * @memberof ExpenseRouter
 * @inner
 * @param {Object} req - Objeto de solicitud de Express.
 * @param {Object} res - Objeto de respuesta de Express.
 */
router.patch(PATCH, validateFirebaseToken, expenseController.updateExpense);

/**
 * Ruta para filtrar gastos por campo y valor.
 *
 * @name POST/api/expenses/filters
 * @function
 * @memberof ExpenseRouter
 * @inner
 * @param {Object} req - Objeto de solicitud de Express.
 * @param {Object} res - Objeto de respuesta de Express.
 */
router.post(
  `${PATCH}/filters`,
  validateFirebaseToken,
  expenseController.filterExpenses
);

module.exports = router;
